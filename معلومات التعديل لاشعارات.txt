<script>
    // تحديد وضع الصفحة: edit للمعلم, view للطلاب
    const urlParams = new URLSearchParams(window.location.search);
    const mode = urlParams.get('mode'); // mode=edit   أو mode=view أو بدونها تعتبر view

    let isAdmin = (mode === 'edit'); // إذا وضعت ?mode=edit في الرابط تصبح مديراً

    // عرض شارة الوضع
    const badge = document.getElementById('modeBadge');
    if (isAdmin) {
        badge.innerHTML = '<i class="fas fa-user-cog"></i> وضع الإدارة (يمكنك التعديل)';
        badge.style.background = "#dc3545";
        badge.style.color = "white";
    } else {
        badge.innerHTML = '<i class="fas fa-eye"></i> وضع الطالب (عرض فقط)';
        badge.style.background = "#17a2b8";
        badge.style.color = "white";
    }

    // البيانات الافتراضية (ستحفظ في localStorage عند التعديل)
    let notifications = [
        { id: 1, title: "?? تذكير: اختبار منتصف الفصل", content: "موعد اختبار منتصف الفصل لمادة قواعد البيانات: الأربعاء القادم الساعة 10 صباحاً." },
        { id: 2, title: "?? نتائج الاختبار التجريبي", content: "تم إعلان نتائج الاختبار التجريبي. درجة النجاح 70%. يرجى مراجعة حساباتكم." },
        { id: 3, title: "?? تحديث المنصة", content: "تم إضافة ميزة مراجعة الإجابات بعد الاختبار. نوصي بتحديث التطبيق." },
        { id: 4, title: "? موعد التسليم", content: "آخر موعد للتسجيل في الاختبار النهائي هو 20 أبريل 2025." },
        { id: 5, title: "?? محاضرة مراجعة", content: "محاضرة مراجعة شاملة يوم السبت من 2 إلى 4 مساءً (قاعة A101)." }
    ];

    // تحميل البيانات المخزنة (حتى يرى الطالب آخر تعديلات المعلم)
    function loadNotifications() {
        const stored = localStorage.getItem("exam_app_notifications");
        if (stored) {
            notifications = JSON.parse(stored);
        }
        render();
    }

    // عرض الإشعارات حسب الوضع
    function render() {
        const container = document.getElementById("notificationsContainer");
        container.innerHTML = "";
        
        notifications.forEach((notif, idx) => {
            const card = document.createElement("div");
            card.className = "card";
            
            if (isAdmin) {
                // وضع المعلم: يعرض حقول editable
                card.innerHTML = `
                    <div class="card-title"><i class="fas fa-pen"></i> <input type="text" id="title-${notif.id}" value="${escapeHtml(notif.title)}" style="font-weight:bold; border:1px solid #ccc; background:#f9f9f9;"></div>
                    <textarea id="content-${notif.id}" rows="3">${escapeHtml(notif.content)}</textarea>
                    <div class="admin-controls">
                        <button class="save-single" data-id="${notif.id}"><i class="fas fa-save"></i> حفظ هذا الإشعار</button>
                    </div>
                `;
            } else {
                // وضع الطالب: عرض نص فقط
                card.innerHTML = `
                    <div class="card-title"><i class="fas fa-bell"></i> ${escapeHtml(notif.title)}</div>
                    <div class="card-content">${escapeHtml(notif.content)}</div>
                `;
            }
            container.appendChild(card);
        });

        if (isAdmin) {
            // إضافة زر حفظ الكل في الأسفل
            const saveAllBtn = document.createElement("button");
            saveAllBtn.className = "save-all-btn";
            saveAllBtn.innerHTML = '<i class="fas fa-save"></i> حفظ جميع التغييرات نهائياً (سيظهر للطلاب)';
            saveAllBtn.onclick = saveAllChanges;
            container.appendChild(saveAllBtn);

            // ربط أزرار الحفظ الفردية
            document.querySelectorAll('.save-single').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    const id = parseInt(btn.getAttribute('data-id'));
                    saveSingleNotification(id);
                });
            });
        }
    }

    function saveSingleNotification(id) {
        const newTitle = document.getElementById(`title-${id}`).value;
        const newContent = document.getElementById(`content-${id}`).value;
        const index = notifications.findIndex(n => n.id === id);
        if (index !== -1) {
            notifications[index].title = newTitle;
            notifications[index].content = newContent;
            localStorage.setItem("exam_app_notifications", JSON.stringify(notifications));
            showToast("تم حفظ هذا الإشعار بنجاح");
        }
    }

    function saveAllChanges() {
        notifications.forEach(notif => {
            const titleEl = document.getElementById(`title-${notif.id}`);
            const contentEl = document.getElementById(`content-${notif.id}`);
            if (titleEl && contentEl) {
                notif.title = titleEl.value;
                notif.content = contentEl.value;
            }
        });
        localStorage.setItem("exam_app_notifications", JSON.stringify(notifications));
        showToast("? تم حفظ جميع الإشعارات! الطلاب سيرون النسخة المحدثة عند تحديث الصفحة.");
    }

    function escapeHtml(str) {
        return str.replace(/[&<>]/g, function(m) {
            if (m === '&') return '&amp;';
            if (m === '<') return '&lt;';
            if (m === '>') return '&gt;';
            return m;
        });
    }

    function showToast(msg) {
        const toast = document.getElementById('toastMsg');
        toast.innerText = msg;
        toast.style.display = "block";
        setTimeout(() => { toast.style.display = "none"; }, 2000);
    }

    loadNotifications();
</script>

كيف تستخدم هذا الحل؟
انسخ الكود أعلاه بالكامل وألصقه في ملف جديد باسم notifications.html.

افتح الملف على متصفح جهازك (جهاز المعلم).

للتعديل كمعلم: أضف ?mode=edit إلى نهاية الرابط.

مثال: C:\Users\...\notifications.html?mode=edit

ستظهر لك أزرار وحقول التعديل. قم بتغيير النصوص كما تريد، ثم اضغط "حفظ جميع التغييرات".

لعرض الصفحة للطلاب: افتح نفس الملف ولكن بدون ?mode=edit.

مثال: C:\Users\...\notifications.html

أو انسخ مسار الملف وأرسله للطلاب، أو اعرضه على شاشة القاعة.

الطلاب سيرون الإشعارات بشكل جميل ونظيف بدون أي أزرار تعديل.

ملاحظة أمان إضافية:
إذا كنت قلقاً من أن الطالب الذكي قد يضيف ?mode=edit بنفسه ويعدل الإشعارات (على جهازه فقط)، فلا تقلق، لأن تعديله سيحفظ في متصفحه هو فقط ولن يؤثر على ما يراه باقي الطلاب. ولكن إذا أردت منع ذلك تماماً، يمكنك ببساطة إزالة شرط ?mode=edit من الرابط الذي تعطيه للطلاب، أو استخدام تطبيق Notepad لحذف كود الإدارة نهائياً من نسخة الطلاب.

هذا هو الحل الأمثل والعملي الذي يفصل بين صلاحيات المعلم والطالب في صفحة واحدة أنيقة ومتوافقة مع الجوال.
